# Queue se agla link uthao (Lekin abhi hatao mat, shift baad me karenge)

try:
    # Audio + Video stream setup
    stream = MediaStream(
        next_media["url"],
        audio_parameters=MediaStream.AudioParameters(bitrate=48000),
        video_parameters=MediaStream.VideoParameters(width=1280, height=720, fps=30)
    )
    
    # Stream change karo bina call disconnect kiye
    await call_py.change_stream(chat_id, stream)
    
except Exception as e:
    print(f"Play karne me error aaya: {e}")
    # Agar is gane me error aaya, to isko skip karke agle par jao
    if QUEUE[chat_id]:
        QUEUE[chat_id].pop(0)
    await play_next_song(chat_id)
